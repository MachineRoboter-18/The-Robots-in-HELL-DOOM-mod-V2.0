class super_armor_core : CustomInventory replaces Megasphere
{
	Default
	{
		+COUNTITEM
		+INVENTORY.ALWAYSPICKUP
		+INVENTORY.ISHEALTH
		+INVENTORY.ISARMOR
		Inventory.PickupMessage "$GOTMSPHERE";
		Inventory.PickupSound "misc/p_pkup";
	}
	States
	{
	Spawn:
		i5ss A -1 BRIGHT light("super_energon_core_light");
		stop;
	Pickup:
		TNT1 A 0 A_GiveInventory("super_armor", 1);
		TNT1 A 0 A_GiveInventory("super_health", 1);
		Stop;
	}
}
