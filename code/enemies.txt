/* ==============================================================================
 * DOCUMENTACIÓN DE IA TÁCTICA: FACCIÓN "OBLITERATED" (Alpha 1-02)
 * ==============================================================================
 * RESUMEN DE ARQUITECTURA:
 * Este archivo define el comportamiento fundamental de los enemigos "Obliterated".
 * Rompe con el paradigma clásico del 'A_Chase' de Doom para implementar una 
 * máquina de estados basada en evaluación constante de distancias (Distance2D) y 
 * comprobación de línea de visión (CheckSight).
 * * CONCEPTOS CLAVE:
 * 1. Conciencia Espacial: El enemigo no solo camina hacia el objetivo; evalúa 
 * continuamente si está en rango óptimo de fuego, rango de retirada, o si 
 * debe iniciar una maniobra de flanqueo.
 * 2. Cerebro Post-Ataque (OBLaftershot): A diferencia de los monstruos de Doom 
 * que se quedan quietos tras disparar, esta función obliga al actor a tomar 
 * una decisión táctica inmediata tras cada ráfaga o disparo, basándose en 
 * la proximidad del jugador.
 * 3. Evasión y Recuperación (retreat_angle): Durante el reposicionamiento, el 
 * actor memoriza su vector de movimiento, gira para disparar, y recupera su 
 * trayectoria evasiva original, simulando tácticas de infantería (Strafing).
 *
 * DICCIONARIO DE FUNCIONES (ACTION VOIDS):
 * - OBLchase: Reemplazo del 'A_Chase'. Decide entre patrullar, perseguir 
 * agresivamente o disparar dependiendo de la visión y la distancia.
 * - OBLchaseplayer: Persecución enfocada cuando hay confirmación visual clara.
 * - OBLcheckproximity: Evaluador de rango crítico. Fuerza el combate cuerpo a 
 * cuerpo (melee) o la retirada (retreat) si el jugador invade su espacio.
 * - OBLaftershot: El núcleo lógico. Decide el siguiente estado (huir, evadir, 
 * o disparar en movimiento) usando el parámetro 'strafe_fire' como booleano 
 * para definir si la clase actual es capaz de disparar mientras corre.
 * - OBLpatrol: Deambulación inactiva hasta detectar al objetivo.
 *
 * HERENCIA:
 * La clase 'obliterated' es la base (abstracta). Monstruos específicos como el 
 * 'obliterated_shotgun' heredan de ella, ajustando parámetros de salud, velocidad, 
 * y limitando su capacidad de fuego en movimiento (strafe_fire = false) para 
 * equilibrar su letalidad (hitscan).
 * ============================================================================== */

class obliterated : actor
{
	double retreat_angle;
	
	default
	{
		health 50;
		radius 18;
		height 56;
		mass 150;
		speed 12;
		fastspeed 13;
		damagefactor "fire", 3.0;
		damagefactor "none", 1.2;
		+floorclip
		+pushable
		+countkill
		+fixmapthingpos
		-nogravity
		PainChance 256;
		monster;
		
		tag "obliterated base";
		
		PainSound "OBL_pain";
		DeathSound "OBL_death";
		AttackSound "obl_attack";
		ActiveSound "obl_spawn";

		
	}
	
	action void OBLchase(double Fire_Range = 400 ) //perseguir
	{
		if (target == null) //si no encuentra a nadie entonces vuelve a waits
		{
			SetStateLabel("waits");
			return;
		}
		
		if (target.player != null) //si el target es un jugador
		{
			double dist = Distance2D(target);
			
			if (CheckSight(target)) //revisa si el target es visible
			{
				if (dist <= Fire_Range)//si la distancia es menor o igual a 600
				{
					SetStateLabel("missile");
					return;
				}
				else
				{
					SetStateLabel("chase_player");
					return;
				}
			}
			else
			{
				a_chase(); //sigue moviendose aunque no vea
				return;
			}
		}
		else
		{
			a_cleartarget();//si el target es un monstruo lo ignora
			SetStateLabel("waits");
			return;
		}
	}
	action void OBLchaseplayer(double Fire_Range = 350) //preseguir al jugador
	{
		if (target == null) return;
		
		double dist = Distance2D(target);
		a_chase();
		
		if (dist <= Fire_Range && CheckSight(target))
		{
			SetStateLabel("missile");
			return;
		}
	}
	action void OBLcheckproximity(double melee_range_attack = 64, double retreat_range = 120, bool do_melee = true) //revisa proximidad
	{
		if (target == null) return;
		
		double dist = Distance2D(target);
		
		if (dist < melee_range_attack && do_melee)
		{
			SetStateLabel("melee");
			return;
		}
		else if (dist < retreat_range)
		{
			SetStateLabel("retreat");
			return;
		}
	}
	action void OBLaftershot(double retreat_range = 100, double repositioning_range = 120, double Fire_Range = 350, bool strafe_fire = true) //luego de disparar
	{
		if (target == null)
		{
			SetStateLabel("see");
			return;
		}
		
		double dist = Distance2D(target);
		
		if (dist < retreat_range)
		{
			SetStateLabel("retreat");
			return;
		}
		else if (dist < fire_range && dist > repositioning_range && strafe_fire)
		{
			setstatelabel("repositioning_fire");
			return;
		}
		else if (dist > repositioning_range)
		{
			SetStateLabel("repositioning");
			return;
		}
		else
		{
			SetStateLabel("see");
			return;
		}
	}
	action void OBLpatrol()
	{
		if (target != null && target.player != null && CheckSight(target))
		{
			SetStateLabel("chase_player");
			return;
		}
		
		a_chase();
	}
	action void Punch()//puñetaso melee
	{
		int damage = 10;
		double range = 64;
		double angle = 5;
		
		A_CustomMeleeAttack(12,"punch","fall_punch");
		a_recoil(-0.5);
	}
	
	
	states
	{
	spawn:
		poss a 10 a_look();
		loop;
		
	waits:
		poss a 10 a_look();
		poss a 35 OBLpatrol();
		loop;
		
	patrol:
		poss abcd 4 OBLchase();
		goto waits;
		
	see:
		poss abcd 4 OBLchase();
		loop;
		
	chase_player:
		poss abcd 3 OBLchaseplayer();
		loop;
		
	repositioning:
		poss a 4 a_setangle(angle + random(-90, 90));
		poss bcdabcdabcda 4 a_recoil(-3);
		goto see;
	
	repositioning_fire:
		poss a 4 {
			a_setangle(angle + random(-90, 90));
			retreat_angle = angle;
		}
		poss bcd 4 a_recoil(-3);
		poss e 1 a_facetarget();
		poss f 2 bright a_posattack();
		poss e 1 a_setangle(retreat_angle);
		poss abcd 4 a_recoil(-3);
		poss e 1 a_facetarget();
		poss f 2 bright a_posattack();
		poss e 1 a_setangle(retreat_angle);
		poss abcd 4 a_recoil(-3);
		goto see;
		
	retreat:
		poss a 4 a_recoil(1);
		goto see;
		
	melee:
		poss e 4 a_facetarget();
		poss f 3 A_CustomMeleeAttack(12);
		poss e 4 OBLaftershot();
		goto see;
		
	missile:
		poss e 8 a_facetarget();
		poss f 4 bright a_posattack();
		poss e 4 OBLaftershot();
		goto see;
		
	pain:
		poss G 3;
		poss G 3 a_pain();
		goto see;
		
	death:
		POSS H 5;
		POSS I 5 A_Scream;
		POSS J 5 A_NoBlocking;
		POSS K 5;
		POSS L -1;
		Stop;
	XDeath:
		POSS M 5;
		POSS N 5 A_XScream;
		POSS O 5 A_NoBlocking;
		POSS PQRST 5;
		POSS U -1;
		Stop;
	Raise:
		POSS K 5;
		POSS JIH 5;
		Goto See;
	}
}

		loop;
		
	chase_player:
		obsw abcd 2 OBLchaseplayer(250);
		loop;

		obsw a 3 a_setangle(angle + random(-90,90));
		obsw bcdabcd 3 a_recoil(-3);
		goto see;
		
	retreat:
		obsw a 3 a_recoil(1);
		goto see;
		
	melee:
		obsg a 2 a_facetarget();
		obsg b 3 A_CustomMeleeAttack(12);
		obsg a 2 OBLaftershot(150,160,280,false);
		goto see;
		
	missile:
		obsu a 6;
		obsu b 5 a_facetarget();
		goto actuallymissile;

		obsf a 8 bright A_SposAttackUseAtkSound();
		obsf b 5;
		obsg aba 4;
		obsu b 2 OBLaftershot(150,160,280,false);
		goto see;
		
	pain:
		obsp a 6 A_Pain();
		obsp b 6 a_recoil(2);
		goto see;

		obsd a 5 {
			A_Scream();

			a_spawnitemex("single_shotgun_shell",0,0,8,0,0,0,0,SXF_NOCHECKPOSITION,0);
		}
		obsd b 5 A_NoBlocking();
		obsd c -1;
		stop;
	}
}

class TRH_imp : actor
{
	default
	{
		health 75;
		speed 10;
		height 56;
		radius 19;
		mass 100;
		monster;
		+floorclip;
		+PUSHABLE;
		+JUMPDOWN;
		+FORCEINFIGHTING;
		+COUNTKILL;
	}
	
	action void a_imp_chase(double max_melee_range = 64, double min_fire_range = 96, double max_fire_range = 512, double min_jump_distance = 48, double max_jump_distance = 128, bool can_jump = true)
	{
		double distXY = Distance2D(target);
		double distZ = pos.z - target.pos.z;
		
		/*
		ahora lo que sigue es:
			*1 verifica si esta en el rango de disparo (si es true intentara disparar)
			*2 verifica si puede saltar si la distZ es mayor a 48 y menor a 128. esto lo podemos manejar gracias a las 2 variables double definidas en la funcion (claro esta si el bool de can_jump es true sino obvia este if)
			la 2 fue desabilitada gracias a la bandera JUMPDOWN
			*3 verifica si esta en el rango de ataque a melee (si es correcto intentara atacar a melee pero si falla y esta a menos de el minimo del rango de disparo procedera a pegar un pequeño dash hacia adelante)
		*/
		if (!target)
		{
			setstatelabel("see");
			return;
		}
		
		if (checksight(target)) //es fundamental porque sino el imp atacaria a la nada, esto aplica si a lo que ataca es tanto jugador como monstruo
		{
			if((distXY >= min_fire_range) && (distXY <= max_fire_range)) //prioridad 1, ante cualquier situacion el imp intentara ser una torreta de proyectiles
			{
				setstatelabel("missile");
				return;
			}
			/*
			if(can_jump && (distZ >= min_jump_distance) && (distZ <= max_jump_distance)) //prioridad 2, si llega a pasar que la distancia vertical del target esta entre estos 2 valores entonces el imp intentara saltar hacia su objetivo
			{
				setstatelabel("jump");
				return;
			}
			*/
			if(distXY <= max_melee_range) //esto es una verificacion, por si luego de saltar las anteriores 2 verificaciones if dan false el imp se preguntara si puede atacar a melee
			{
				setstatelabel("melee");
				return;
			}
			if((distXY >= max_melee_range) && (distXY <= min_fire_range)) //si la anterior verificacion falla por que el jugador esta exactamente en este pequeño valle bacio el imp intentara hacer un dash para intentar acercarse al jugador, esto a efectos practicos nunca o casi nunca funcionara ya que se supone que el jugador se mueve constantemente asi que si logra hacer un dash el imp inmediatamente debera o disparar haciendo su respectiva verificacion o atacar a melee
			{
				setstatelabel("dash");
				return;
			}
		}
		a_chase(); //si todo lo demas falla hace el a_chase con normalidad
	}
	
	action void a_imp_aftershot(bool can_retreat = true, bool can_repositioning = false, double repositioning_range = 256, double retreat_range = 128 ,double max_melee_range = 64 , double min_fire_range = 65, double max_fire_range = 512 )
	{
		/*
			en esta funcion se hace mas o menos lo mismo salvo el hecho de que verifica los valores de nuevo para tomar una decision, ademas que cuenta con casi las mismas variables que la funcion anterior.
		*/
		double distXY = Distance2D(target);
		
		if (checksight(target))
		{
			if( can_repositioning && (distXY >= repositioning_range))
			{
				setstatelabel("repositioning");
				return;
			}
			if( can_retreat && (distXY <= retreat_range) && (distXY >= max_melee_range))
			{
				setstatelabel("retreat");
				return;
			}
			if( (distXY >= min_fire_range) && (distXY <= max_fire_range))
			{
				setstatelabel("missile");
				return;
			}
		}
		SetStateLabel("see");
		return;
	}
	
	action bool a_check_floor_repositioning(double angle_to_rotate = 45, double dist = 64)
	{
		vector2 checkpos = vec2angle(dist, angle + angle_to_rotate);
		
		if(checkmove(checkpos))
		{
			return true;
		}
		return false;
	}
	action bool a_check_floor_retreat(double dist = 48)
	{
		vector2 backpos = vec2angle(-dist, angle);
		
		if (checkmove(backpos))
		{
		return true;
		}
		return false;
	}
	
	states
	{
	Spawn:
		TROO AB 10 A_Look;
		Loop;
	See:
		TROO AABBCCDD 3 a_imp_chase();
		Loop;
	Melee:
	Missile:
		TROO EF 8 A_FaceTarget;
		TROO G 5 A_TroopAttack ;
		troo e 1 a_imp_aftershot();
		Goto See;
		
	retreat:
		troo a 1 A_FaceTarget();
		troo a 0{
			if(a_check_floor_retreat()){ return resolvestate("null");}
			else { return resolvestate("see"); }
		}
		troo ba 3 a_recoil(2);
		goto see;
		
	repositioning:
		troo a 2 A_FaceTarget();
		troo a 0{
			double new_angle = randompick(-90,90);
			if(a_check_floor_repositioning(new_angle, 64)){ a_setangle(angle + new_angle); return resolvestate("null");}
			else {return resolvestate("see");}
			
		}
		troo abab 4 a_recoil(-3);
		goto see;
		
		
	/*//desabilitado por bandera JUMPDOWN	
	jump:
		troo n 8 {
			vel.z = 6;
			vector2 direction = angletovector(angle, 2.0);
			vel.x = direction.x;
			vel.y = direction.y;
		}
		troo hhhh 8;
		goto see;
	*/	
	dash:
		troo a 2 A_FaceTarget();
		troo b 6 a_recoil(-5);
		goto see;
		
	Pain:
		TROO H 2;
		TROO H 2 A_Pain;
		Goto See;
	Death:
		TROO I 8;
		TROO J 8 A_Scream;
		TROO K 6;
		TROO L 6 A_NoBlocking;
		TROO M -1;
		Stop;
	XDeath:
		TROO N 5;
		TROO O 5 A_XScream;
		TROO P 5;
		TROO Q 5 A_NoBlocking;
		TROO RST 5;
		TROO U -1;
		Stop;
	Raise:
		TROO ML 8;
		TROO KJI 6;
		Goto See;
	}
}
