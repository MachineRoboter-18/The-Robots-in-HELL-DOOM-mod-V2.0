class Cy_orbitary_saw : weapon_base { Default { Tag "Motosierra Orbital"; } States { Spawn: TNT1 A -1; Stop; Select: TNT1 A 1 A_Raise; Loop; Deselect: TNT1 A 1 A_Lower; Loop; Ready: TNT1 A 1 A_WeaponReady; Loop; Fire: TNT1 A 1; Goto Ready; } }
class Cy_Rifle : weapon_base REPLACES PISTOL
{
	Default
	{
		Weapon.SelectionOrder 1900;
		Weapon.AmmoUse 1;
		Weapon.AmmoGive 20;
		Weapon.AmmoType "RIFLE_BULLETS";
		Obituary "$OB_MPPISTOL";
		+WEAPON.WIMPY_WEAPON
		Inventory.Pickupmessage "$PICKUP_PISTOL_DROPPED";
		Tag "$TAG_PISTOL";
	}
	States
	{
	Ready:
		PISG A 1 A_WeaponReady;
		Loop;
	Deselect:
		PISG A 0 A_Lower;
		wait;
	Select:
		PISG A 1 A_Raise;
		Loop;
	Fire:
		PISF A 2 BRIGHT A_FirePistol;
		PISG CD 1;
		pisg e 2 a_fireprojectile("RifleCaseSpawn",1,0,6,0);
		PISG B 4 A_ReFire;
		Goto Ready;

 	Spawn:
		I3RF A -1;
		Stop;
	}
}

		misf b 4;
		MIsF A 4 bright a_firerpg69missile();
		misf b 4;
		MIsF A 4 bright a_firerpg69missile();
		misf bcd 4;
		misf b 8;
		goto ready;

	Spawn:
		i3rk A -1;
		Stop;
	}
}

class Cy_traffic_cone : weapon_base 
{
	Default
	{
		Weapon.SelectionOrder 3700;
		Weapon.Kickback 100;
		Obituary "$OB_MPFIST";
		Tag "$TAG_FIST";
		+WEAPON.WIMPY_WEAPON
		+WEAPON.MELEEWEAPON
	}
	States
	{
	Ready:
		TRCG A 1 A_WeaponReady;
		Loop;
	Deselect:
		TRCG A 0 A_Lower;
		WAIT;
	Select:
		TRCG A 1 A_Raise;
		Loop;
	Fire:
		TRCF AB 3;
		TRCF c 7 a_punch();
		trcf ba 2;
		tnt1 a 0 a_refire();
		goto ready;
		
	Spawn:
		i3cv b -1;
		stop;
	}
}
